<Canvas>

		{() => {
			const [value, setValue] = useState('10:30:0:0');
			return (
				<Timepicker
					id="time"
					value={value}
					onChange={(newValue) => {
						action()(newValue);
						setValue(newValue);

	<Story name='Required'>

						action()(newValue);
						setValue(newValue);
					}}
					required
				/>
			);
		}}
	</Story>
	<Story name='Stepped'>
		{() => {
			const [value, setValue] = useState('10:30:0:0');

				<Timepicker
					id="time"
					label="Time"
					value={value}
					step={900}
					onChange={(newValue) => {
